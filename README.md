    <add key="ConnectionString" value="Provider=SQLOLEDB;D͟a͟t͟a͟ ͟S͟o͟u͟r͟c͟e͟=͟T͟r͟u͟n͟g͟T͟a͟n͟\͟S͟Q͟L͟E͟X͟P͟R͟E͟S͟S͟0͟1͟;database=shopquanao;integrated security=sspi;charset=utf8"/>
